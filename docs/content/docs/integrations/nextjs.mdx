## Installation

```bash
npm install @invect/core @invect/nextjs
```

## Setup

### API Routes

Create a catch-all API route for Invect:

```ts title="app/api/invect/[...invect]/route.ts"
import { createInvectHandler } from '@invect/nextjs';

const handler = createInvectHandler({
  database: {
    type: 'sqlite',
    connectionString: process.env.DATABASE_URL || 'file:./dev.db',
    id: 'main',
  },
});

export { handler as GET, handler as POST, handler as PUT, handler as DELETE, handler as PATCH };
```

### Frontend Component

In a client page or layout, mount the Invect component:

```tsx title="app/workflows/page.tsx"
'use client';

import { Invect } from '@invect/ui';
import '@invect/ui/styles';

export default function WorkflowsPage() {
  return <Invect apiBaseUrl="/api/invect" />;
}
```

## Server Actions

You can also use the core directly in server actions:

```ts title="app/actions/workflows.ts"
'use server';

import { createInvect, type InvectInstance } from '@invect/core';

let invect: InvectInstance | null = null;
let initPromise: Promise<InvectInstance> | null = null;

function getInvect(): Promise<InvectInstance> {
  if (invect) return Promise.resolve(invect);
  if (!initPromise) {
    initPromise = createInvect({
      database: {
        type: 'sqlite',
        connectionString: process.env.DATABASE_URL || 'file:./dev.db',
        id: 'main',
      },
    }).then((instance) => {
      invect = instance;
      return instance;
    });
  }
  return initPromise;
}

export async function executeWorkflow(flowId: string, inputs: Record<string, unknown>) {
  const core = await getInvect();
  return core.runs.start(flowId, inputs);
}

export async function listWorkflows() {
  const core = await getInvect();
  return core.flows.list();
}
```

## Triggering Flows from Server Components

```tsx title="app/dashboard/page.tsx"
import { executeWorkflow, listWorkflows } from '@/app/actions/workflows';

export default async function Dashboard() {
  const flows = await listWorkflows();

  return (
    <div>
      <h1>Workflows</h1>
      {flows.map(flow => (
        <div key={flow.id}>
          <h2>{flow.name}</h2>
          <form action={async () => {
            'use server';
            await executeWorkflow(flow.id, {});
          }}>
            <button type="submit">Run</button>
          </form>
        </div>
      ))}
    </div>
  );
}
```
