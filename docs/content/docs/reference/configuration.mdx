---
title: Configuration
description: All options for invect.config.ts.
---

The Invect configuration object is passed to `createInvectRouter()` (Express), `InvectModule.forRoot()` (NestJS), or `createInvectHandler()` (Next.js). Use `defineConfig()` from `@invect/core` for type inference.

```ts title="invect.config.ts"
import { defineConfig } from '@invect/core';

export const invectConfig = defineConfig({
  database: { ... },
  logging: { ... },
  execution: { ... },
  triggers: { ... },
  plugins: [ ... ],
  defaultCredentials: [ ... ],
});
```

## `database` (required)

| Field | Type | Default | Description |
|-------|------|---------|-------------|
| `connectionString` | `string` | — | Database URL. SQLite: `file:./dev.db`. PostgreSQL: `postgresql://user:pass@host:port/db`. MySQL: `mysql://user:pass@host:port/db`. |
| `type` | `'sqlite' \| 'postgresql' \| 'mysql'` | — | Database dialect. |
| `name` | `string` | — | Optional human-readable name. |
| `driver` | see below | auto-detected | Database driver to use. |

### Supported drivers

| Database | Drivers | Default |
|----------|---------|---------|
| PostgreSQL | `postgres`, `pg`, `neon-serverless` | `postgres` |
| SQLite | `better-sqlite3`, `libsql` | `better-sqlite3` |
| MySQL | `mysql2` | `mysql2` |

```ts
// SQLite (simplest for local dev)
database: {
  type: 'sqlite',
  connectionString: 'file:./dev.db',
}

// PostgreSQL
database: {
  type: 'postgresql',
  connectionString: 'postgresql://user:pass@localhost:5432/invect',
}

// PostgreSQL with Neon serverless driver
database: {
  type: 'postgresql',
  connectionString: process.env.DATABASE_URL,
  driver: 'neon-serverless',
}

// MySQL
database: {
  type: 'mysql',
  connectionString: 'mysql://user:pass@localhost:3306/invect',
}
```

## `logging` (optional)

| Field | Type | Default | Description |
|-------|------|---------|-------------|
| `level` | `'debug' \| 'info' \| 'warn' \| 'error' \| 'silent'` | `'silent'` | Global log level. |
| `scopes` | `Record<string, LogLevel>` | — | Per-scope log level overrides. |

### Available logging scopes

`execution`, `validation`, `batch`, `database`, `node`, `graph`, `credentials`, `ai`, `template`, `renderer`, `flows`, `versions`, `http`

```ts
logging: {
  level: 'info',
  scopes: {
    execution: 'debug',  // Verbose execution logs
    batch: 'silent',     // Suppress batch polling logs
    ai: 'debug',        // Debug AI provider calls
  },
}
```

## `execution` (optional)

| Field | Type | Default | Description |
|-------|------|---------|-------------|
| `defaultTimeout` | `number` | `60000` | Node execution timeout in milliseconds. |
| `maxConcurrentExecutions` | `number` | `10` | Maximum parallel flow runs. |
| `enableTracing` | `boolean` | `true` | Store node-level execution traces. |
| `flowTimeoutMs` | `number` | `600000` | Flow run timeout in milliseconds (10 minutes). |
| `heartbeatIntervalMs` | `number` | `30000` | Status heartbeat interval in milliseconds. |
| `staleRunCheckIntervalMs` | `number` | `60000` | Interval for checking stale flow runs. |

```ts
execution: {
  defaultTimeout: 120000,         // 2 minute node timeout
  maxConcurrentExecutions: 5,     // Limit parallel runs
  flowTimeoutMs: 300_000,         // 5 minute flow timeout
}
```

## `triggers` (optional)

| Field | Type | Default | Description |
|-------|------|---------|-------------|
| `webhookBaseUrl` | `string` | — | Public URL prefix for webhook URLs displayed in the editor. |
| `cronEnabled` | `boolean` | `true` | Enable or disable the cron scheduler. |

```ts
triggers: {
  webhookBaseUrl: 'https://api.myapp.com/invect',
  cronEnabled: true,
}
```

## `plugins` (optional)

An array of plugin instances. Plugins are initialised in order during startup and shut down in reverse order.

```ts
import { userAuth } from '@invect/user-auth';
import { rbacPlugin } from '@invect/rbac';
import { webhooksPlugin } from '@invect/webhooks';

plugins: [
  userAuth({
    betterAuthOptions: { secret: process.env.BETTER_AUTH_SECRET },
    globalAdmins: [{ email: 'admin@example.com', pw: 'changeme', name: 'Admin' }],
  }),
  rbacPlugin(),
  webhooksPlugin({ webhookBaseUrl: 'https://api.myapp.com/invect' }),
]
```

See [Plugin API reference](/docs/reference/plugin-api) for the plugin interface and [How-to guides](/docs/how-to/plugins/add-authentication) for individual plugin setup.

## `defaultCredentials` (optional)

An array of credentials to seed on startup. Credentials are matched by name — existing credentials are not updated.

Each credential has:

| Field | Type | Required | Description |
|-------|------|----------|-------------|
| `name` | `string` | Yes | Unique credential name. |
| `type` | `'llm' \| 'http-api' \| 'database'` | Yes | Credential category. |
| `provider` | `string` | For `llm` | Provider identifier (e.g., `openai`, `anthropic`). |
| `authType` | `string` | Yes | Authentication method (e.g., `apiKey`, `oauth2`). |
| `config` | `Record<string, unknown>` | Yes | Provider-specific config (e.g., `{ apiKey: '...' }`). |
| `description` | `string` | No | Human-readable note. |
| `isShared` | `boolean` | No | Whether the credential is visible to all users. |
| `metadata` | `Record<string, unknown>` | No | Arbitrary metadata. |

```ts
defaultCredentials: [
  {
    name: 'OpenAI',
    type: 'llm',
    provider: 'openai',
    authType: 'apiKey',
    config: { apiKey: process.env.OPENAI_API_KEY },
    isShared: true,
  },
  {
    name: 'Anthropic',
    type: 'llm',
    provider: 'anthropic',
    authType: 'apiKey',
    config: { apiKey: process.env.ANTHROPIC_API_KEY },
    isShared: true,
  },
]
```

## `basePath` (optional)

| Field | Type | Default | Description |
|-------|------|---------|-------------|
| `basePath` | `string` | — | Base URL path prefix for all Invect routes. |

## `logger` (optional)

| Field | Type | Default | Description |
|-------|------|---------|-------------|
| `logger` | `Logger` | Console logger | Custom logger instance. Must implement `info`, `debug`, `warn`, `error` methods. |

## Full example

```ts title="invect.config.ts"
import { defineConfig } from '@invect/core';
import { userAuth } from '@invect/user-auth';
import { rbacPlugin } from '@invect/rbac';
import { webhooksPlugin } from '@invect/webhooks';

export const invectConfig = defineConfig({
  database: {
    type: 'postgresql',
    connectionString: process.env.DATABASE_URL,
  },
  logging: {
    level: 'info',
    scopes: {
      execution: 'debug',
    },
  },
  execution: {
    defaultTimeout: 120000,
    maxConcurrentExecutions: 20,
    flowTimeoutMs: 600_000,
  },
  triggers: {
    webhookBaseUrl: process.env.WEBHOOK_BASE_URL,
    cronEnabled: true,
  },
  plugins: [
    userAuth({
      betterAuthOptions: { secret: process.env.BETTER_AUTH_SECRET },
      globalAdmins: [
        { email: process.env.ADMIN_EMAIL, pw: process.env.ADMIN_PASSWORD, name: 'Admin' },
      ],
    }),
    rbacPlugin(),
    webhooksPlugin({ webhookBaseUrl: process.env.WEBHOOK_BASE_URL }),
  ],
  defaultCredentials: [
    {
      name: 'OpenAI',
      type: 'llm',
      provider: 'openai',
      authType: 'apiKey',
      config: { apiKey: process.env.OPENAI_API_KEY },
      isShared: true,
    },
  ],
});
```
