Plugins extend Invect with custom actions, API endpoints, database tables, lifecycle hooks, and frontend components.

## Plugin structure

```ts
import type { InvectPlugin } from '@invect/core';

export function myPlugin(options?: MyPluginOptions): InvectPlugin {
  return {
    id: 'my-plugin',
    name: 'My Plugin',

    // Custom actions (become flow nodes + agent tools)
    actions: [myAction, anotherAction],

    // Lifecycle hooks
    hooks: {
      beforeFlowRun: async (context) => {
        // Runs before every flow execution
        // Return { cancel: true } to abort, or { inputs } to modify inputs
      },
      afterFlowRun: async (context) => {
        // Runs after every flow completes
      },
      onRequest: async (context) => {
        // Runs before every API request
        // Return { response } to short-circuit
      },
    },

    // Custom API endpoints
    endpoints: [
      {
        method: 'GET',
        path: '/plugins/my-plugin/status',
        handler: async (req, res) => {
          res.json({ status: 'ok' });
        },
      },
    ],

    // Database tables (abstract, dialect-agnostic)
    schema: {
      my_plugin_data: {
        fields: {
          id: { type: 'string', primaryKey: true },
          name: { type: 'string', required: true },
          flowId: { type: 'string', references: { table: 'flows', field: 'id' } },
          createdAt: { type: 'timestamp' },
        },
      },
    },

    // Tables checked at startup
    requiredTables: ['my_plugin_data'],

    // Initialisation
    init: async (ctx) => {
      // ctx.db — database connection
      // ctx.logger — scoped logger
      // ctx.registerTool(definition, executor) — register standalone agent tools
    },

    // Cleanup
    shutdown: async () => {
      // Runs on graceful shutdown
    },
  };
}
```

## Available hooks

| Hook | When | Can short-circuit? |
|------|------|--------------------|
| `beforeFlowRun` | Before flow execution | Yes — `{ cancel: true }` stops the run |
| `afterFlowRun` | After flow completes | No |
| `beforeNodeExecute` | Before each node | Yes — `{ skip: true }` skips it |
| `afterNodeExecute` | After each node | Can override output |
| `onRequest` | Before every API request | Yes — return `{ response }` |
| `onResponse` | After every API response | Can replace response |
| `onAuthorize` | During auth checks | Yes — `{ allowed: true/false }` |

## Schema format

Plugin schemas use an abstract format that's converted to dialect-specific Drizzle files by the CLI:

```ts
schema: {
  my_table: {
    fields: {
      id: { type: 'string', primaryKey: true },
      name: { type: 'string', required: true },
      count: { type: 'integer', default: 0 },
      active: { type: 'boolean', default: true },
      flowId: { type: 'string', references: { table: 'flows', field: 'id' } },
      createdAt: { type: 'timestamp' },
    },
  },
}
```

After adding or changing schema, run:

```bash
pnpm dlx @invect/cli generate
pnpm dlx @invect/cli migrate --push
```

## Register the plugin

```ts title="invect.config.ts"
import { defineConfig } from '@invect/core';
import { myPlugin } from './my-plugin';

export const invectConfig = defineConfig({
  database: { type: 'sqlite', connectionString: 'file:./dev.db' },
  plugins: [myPlugin({ /* options */ })],
});
```
