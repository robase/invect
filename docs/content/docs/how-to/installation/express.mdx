---
title: Set Up with Express
description: Install and configure Invect in an Express application.
---

## Install packages

```bash
pnpm add @invect/core @invect/express better-sqlite3 express cors
```

For PostgreSQL instead of SQLite, replace `better-sqlite3` with `postgres`:

```bash
pnpm add @invect/core @invect/express postgres express cors
```

## Create the config

```ts title="invect.config.ts"
import { defineConfig } from '@invect/core';

export const invectConfig = defineConfig({
  database: {
    type: 'sqlite',
    connectionString: 'file:./dev.db',
  },
  logging: {
    level: 'info',
  },
});
```

See the [Configuration reference](/docs/reference/configuration) for all available options.

## Generate and migrate the database

```bash
pnpm dlx @invect/cli generate
pnpm dlx @invect/cli migrate --push
```

## Mount the router

```ts title="index.ts"
import express from 'express';
import cors from 'cors';
import { createInvectRouter } from '@invect/express';
import { invectConfig } from './invect.config';

const app = express();
app.use(cors());
app.use(express.json());

const invectRouter = await createInvectRouter(invectConfig);
app.use('/invect', invectRouter);

app.listen(3000, () => {
  console.log('Invect API running at http://localhost:3000/invect');
});
```

`createInvectRouter` is an async factory that initialises the database, registers all built-in actions and tools, starts the batch polling service, and returns a standard Express router.

## Add the frontend (optional)

If you want the visual editor, add `@invect/ui` to a React frontend and point it at your Express server:

```tsx
import { Invect } from '@invect/ui';
import '@invect/ui/styles';

<Invect apiBaseUrl="http://localhost:3000/invect" />
```

See [Embed the Editor](/docs/how-to/frontend/embed-the-editor) for full frontend setup.

## Add plugins (optional)

To add authentication and access control:

```bash
pnpm add @invect/user-auth @invect/rbac
```

```ts title="invect.config.ts"
import { defineConfig } from '@invect/core';
import { userAuth } from '@invect/user-auth';
import { rbacPlugin } from '@invect/rbac';

export const invectConfig = defineConfig({
  database: {
    type: 'sqlite',
    connectionString: 'file:./dev.db',
  },
  plugins: [
    userAuth({
      betterAuthOptions: { secret: process.env.BETTER_AUTH_SECRET },
      globalAdmins: [{ email: 'admin@example.com', pw: 'changeme', name: 'Admin' }],
    }),
    rbacPlugin(),
  ],
});
```

After adding plugins with database tables, re-run schema generation:

```bash
pnpm dlx @invect/cli generate
pnpm dlx @invect/cli migrate --push
```

See [Add Authentication](/docs/how-to/plugins/add-authentication) and [Add Access Control](/docs/how-to/plugins/add-access-control) for details.
